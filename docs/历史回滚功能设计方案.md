## 1. 功能需求

### 1.1 核心功能

| 功能 | 说明 |
|------|------|
| **撤销（Undo）** | Ctrl+Z，恢复上一步操作 |
| **重做（Redo）** | Ctrl+Y / Ctrl+Shift+Z，重做已撤销的操作 |
| **历史查看** | 显示历史操作列表（可选） |
| **历史清空** | 清空历史记录（打开新项目时） |
| **历史限制** | 最大历史记录数量限制（如 50 步） |

### 1.2 支持的操作类型

| 操作分类 | 具体操作 |
|---------|---------|
| **节点操作** | 创建节点、删除节点、重命名节点、移动节点、修改节点配置 |
| **画布操作** | 创建连线、删除连线、清空画布、移动画布、缩放画布 |
| **视图操作** | 显示/隐藏面板、切换节点样式 |
| **绘图操作** | 绘制内容（Draw Layer） |
| **批量操作** | 批量移动、批量删除等作为一个历史条目 |

### 1.3 非功能需求

| 需求 | 说明 |
|------|------|
| **性能** | 历史记录操作不应影响主线程响应 |
| **稳定性** | 回滚失败时系统应保持一致状态 |
| **持久化** | 历史记录可选保存到项目中（下次打开可继续回滚） |
| **可扩展** | 容易添加新的可撤销操作 |

## 3. 数据结构

### 3.1 Command 基类

```python
# 文件: ui/core/commands/base.py

from abc import ABC, abstractmethod
from dataclasses import dataclass
from typing import Any, Optional
from enum import Enum


class CommandType(Enum):
    """命令类型"""
    NODE_CREATE = "node.create"
    NODE_DELETE = "node.delete"
    NODE_MOVE = "node.move"
    NODE_RENAME = "node.rename"
    NODE_CONFIG = "node.config"
    EDGE_CREATE = "edge.create"
    EDGE_DELETE = "edge.delete"
    CANVAS_CLEAR = "canvas.clear"
    CANVAS_VIEW = "canvas.view"
    DRAWING = "drawing"
    BATCH = "batch"
    COMPOUND = "compound"


@dataclass
class CommandResult:
    """命令执行结果"""
    success: bool
    message: str = ""
    error: Optional[Exception] = None


class Command(ABC):
    """命令基类"""
    
    def __init__(self, command_type: CommandType, description: str = ""):
        self.command_type = command_type
        self.description = description
        self.timestamp = None  # 执行时间戳
        self.executed = False  # 是否已执行
        self.can_undo = True  # 是否可撤销
    
    @abstractmethod
    def execute(self) -> CommandResult:
        """执行命令"""
        pass
    
    @abstractmethod
    def undo(self) -> CommandResult:
        """撤销命令"""
        pass
    
    @abstractmethod
    def redo(self) -> CommandResult:
        """重做命令（默认调用 execute）"""
        pass
    
    def to_dict(self) -> dict:
        """序列化为字典（用于持久化）"""
        return {
            "type": self.command_type.value,
            "description": self.description,
            "timestamp": self.timestamp,
            "can_undo": self.can_undo,
            "data": self._serialize_data()
        }
    
    @classmethod
    def from_dict(cls, data: dict) -> 'Command':
        """从字典反序列化"""
        pass
    
    @abstractmethod
    def _serialize_data(self) -> dict:
        """序列化命令数据"""
        pass
```

### 3.2 HistoryManager 状态

```python
# 文件: ui/core/commands/history_manager.py

from dataclasses import dataclass, field
from typing import List, Optional
from collections import deque


@dataclass
class HistoryState:
    """历史记录状态"""
    undo_stack: List[Command] = field(default_factory=list)
    redo_stack: List[Command] = field(default_factory=list)
    current_index: int = -1
    max_history: int = 50  # 最大历史记录数
    is_recording: bool = True  # 是否正在记录
    
    def get_can_undo(self) -> bool:
        return len(self.undo_stack) > 0
    
    def get_can_redo(self) -> bool:
        return len(self.redo_stack) > 0
    
    def get_undo_description(self) -> Optional[str]:
        if self.undo_stack:
            return self.undo_stack[-1].description
        return None
    
    def get_redo_description(self) -> Optional[str]:
        if self.redo_stack:
            return self.redo_stack[-1].description
        return None
```

## 4. 核心模块设计

### 4.1 HistoryManager（历史管理器）

```python
# 文件: ui/core/commands/history_manager.py

import os
import json
import time
from typing import Optional, List
from PyQt6.QtCore import QObject, pyqtSignal
from ui.core.logger import logger
from ui.core.event_bus import event_bus


class HistoryManager(QObject):
    """历史记录管理器 - 单例模式"""
    
    # 信号
    history_changed = pyqtSignal()  # 历史变更
    can_undo_changed = pyqtSignal(bool)  # 是否可撤销状态变更
    can_redo_changed = pyqtSignal(bool)  # 是否可重做状态变更
    
    _instance = None
    
    def __new__(cls):
        if cls._instance is None:
            cls._instance = super().__new__(cls)
            cls._instance._initialized = False
        return cls._instance
    
    def __init__(self):
        if self._initialized:
            return
        
        super().__init__()
        self.state = HistoryState()
        self._initialized = True
        
        # 注册到 EventBus
        event_bus.subscribe('project.opened', self._on_project_opened)
        event_bus.subscribe('project.closed', self._on_project_closed)
    
    def execute_command(self, command: Command) -> CommandResult:
        """执行命令并记录"""
        if not self.state.is_recording:
            return command.execute()
        
        # 执行命令
        result = command.execute()
        if not result.success:
            return result
        
        # 记录时间戳
        command.timestamp = time.time()
        command.executed = True
        
        # 添加到撤销栈
        self.state.undo_stack.append(command)
        
        # 清空重做栈
        self.state.redo_stack.clear()
        
        # 限制最大历史记录数
        if len(self.state.undo_stack) > self.state.max_history:
            self.state.undo_stack.pop(0)
        
        # 发送变更信号
        self._emit_change_signals()
        
        logger.info(f"Command executed: {command.description}")
        return result
    
    def undo(self) -> CommandResult:
        """撤销"""
        if not self.state.get_can_undo():
            return CommandResult(False, "没有可撤销的操作")
        
        command = self.state.undo_stack.pop()
        result = command.undo()
        
        if result.success:
            self.state.redo_stack.append(command)
            self._emit_change_signals()
            logger.info(f"Undo: {command.description}")
        
        return result
    
    def redo(self) -> CommandResult:
        """重做"""
        if not self.state.get_can_redo():
            return CommandResult(False, "没有可重做的操作")
        
        command = self.state.redo_stack.pop()
        result = command.redo()
        
        if result.success:
            self.state.undo_stack.append(command)
            self._emit_change_signals()
            logger.info(f"Redo: {command.description}")
        
        return result
    
    def clear_history(self):
        """清空历史记录"""
        self.state.undo_stack.clear()
        self.state.redo_stack.clear()
        self._emit_change_signals()
        logger.info("History cleared")
    
    def can_undo(self) -> bool:
        return self.state.get_can_undo()
    
    def can_redo(self) -> bool:
        return self.state.get_can_redo()
    
    def get_undo_history(self, limit: int = 10) -> List[Command]:
        """获取撤销历史"""
        return self.state.undo_stack[-limit:] if limit > 0 else self.state.undo_stack.copy()
    
    def get_redo_history(self, limit: int = 10) -> List[Command]:
        """获取重做历史"""
        return self.state.redo_stack[-limit:] if limit > 0 else self.state.redo_stack.copy()
    
    def save_history(self, project_path: str):
        """保存历史到项目文件（可选）"""
        if not project_path:
            return
        
        history_file = os.path.join(project_path, "history.json")
        data = {
            "undo_stack": [cmd.to_dict() for cmd in self.state.undo_stack],
            "redo_stack": [cmd.to_dict() for cmd in self.state.redo_stack]
        }
        
        try:
            with open(history_file, "w", encoding="utf-8") as f:
                json.dump(data, f, indent=2, ensure_ascii=False)
            logger.info(f"History saved to {history_file}")
        except Exception as e:
            logger.error(f"Failed to save history: {e}")
    
    def load_history(self, project_path: str):
        """从项目文件加载历史（可选）"""
        # TODO: 实现历史加载
        pass
    
    def begin_macro(self, description: str) -> 'MacroCommand':
        """开始宏（批量操作）"""
        from .compound_commands import MacroCommand
        return MacroCommand(description)
    
    def pause_recording(self):
        """暂停记录"""
        self.state.is_recording = False
    
    def resume_recording(self):
        """恢复记录"""
        self.state.is_recording = True
    
    def _emit_change_signals(self):
        """发送变更信号"""
        self.history_changed.emit()
        self.can_undo_changed.emit(self.can_undo())
        self.can_redo_changed.emit(self.can_redo())
    
    def _on_project_opened(self, project_path: str):
        """项目打开时清空历史"""
        self.clear_history()
    
    def _on_project_closed(self):
        """项目关闭时清空历史"""
        self.clear_history()
```

### 4.3 复合命令（Macro）

```python
# 文件: ui/core/commands/compound_commands.py

from .base import Command, CommandType, CommandResult
from typing import List


class MacroCommand(Command):
    """宏命令 - 组合多个命令作为一个历史条目"""
    
    def __init__(self, description: str = ""):
        super().__init__(CommandType.COMPOUND, description or "批量操作")
        self._commands: List[Command] = []
    
    def add_command(self, command: Command):
        """添加子命令"""
        self._commands.append(command)
    
    def execute(self) -> CommandResult:
        for cmd in self._commands:
            result = cmd.execute()
            if not result.success:
                return result
        
        if not self.description:
            self.description = f"{len(self._commands)} 个操作"
        
        return CommandResult(True)
    
    def undo(self) -> CommandResult:
        # 逆序撤销
        for cmd in reversed(self._commands):
            result = cmd.undo()
            if not result.success:
                return result
        
        return CommandResult(True)
    
    def redo(self) -> CommandResult:
        return self.execute()
    
    def _serialize_data(self) -> dict:
        return {
            "commands": [cmd.to_dict() for cmd in self._commands]
        }
```

```python
# 在 MainWindow 中添加菜单和快捷键
from ui.core.commands.history_manager import history_manager
from ui.core.i18n import t


class MainWindow(...):
    # ... 现有代码 ...
    
    def init_ui(self):
        # ... 现有代码 ...
        
        # 注册撤销/重做快捷键
        self._setup_undo_redo_shortcuts()
        
        # 连接信号更新菜单状态
        history_manager.can_undo_changed.connect(self._update_undo_menu)
        history_manager.can_redo_changed.connect(self._update_redo_menu)
    
    def _setup_undo_redo_shortcuts(self):
        """设置快捷键"""
        # Undo: Ctrl+Z
        undo_action = QAction(t("k_edit_undo"), self)
        undo_action.setShortcut("Ctrl+Z")
        undo_action.triggered.connect(history_manager.undo)
        self.addAction(undo_action)
        self._undo_action = undo_action
        
        # Redo: Ctrl+Y 或 Ctrl+Shift+Z
        redo_action = QAction(t("k_edit_redo"), self)
        redo_action.setShortcut("Ctrl+Y")
        redo_action.triggered.connect(history_manager.redo)
        self.addAction(redo_action)
        self._redo_action = redo_action
    
    def _update_undo_menu(self, can_undo: bool):
        """更新撤销菜单状态"""
        if hasattr(self, '_undo_action'):
            self._undo_action.setEnabled(can_undo)
            if can_undo:
                desc = history_manager.state.get_undo_description()
                self._undo_action.setText(f"{t('k_edit_undo')} - {desc}" if desc else t('k_edit_undo'))
    
    def _update_redo_menu(self, can_redo: bool):
        """更新重做菜单状态"""
        if hasattr(self, '_redo_action'):
            self._redo_action.setEnabled(can_redo)
            if can_redo:
                desc = history_manager.state.get_redo_description()
                self._redo_action.setText(f"{t('k_edit_redo')} - {desc}" if desc else t('k_edit_redo'))
```

### 7.1 单元测试

```python
# 文件: tests/test_history.py

import pytest
from ui.core.commands.history_manager import HistoryManager
from ui.core.commands.base import Command, CommandType, CommandResult


class MockCommand(Command):
    """测试用 mock 命令"""
    def __init__(self):
        super().__init__(CommandType.NODE_CREATE, "Test Command")
        self.execute_called = 0
        self.undo_called = 0
        self.redo_called = 0
    
    def execute(self):
        self.execute_called += 1
        return CommandResult(True)
    
    def undo(self):
        self.undo_called += 1
        return CommandResult(True)
    
    def redo(self):
        self.redo_called += 1
        return CommandResult(True)
    
    def _serialize_data(self):
        return {}


class TestHistoryManager:
    """历史管理器测试"""
    
    def setup_method(self):
        self.manager = HistoryManager()
        self.manager.clear_history()
    
    def test_execute_and_undo(self):
        cmd = MockCommand()
        self.manager.execute_command(cmd)
        
        assert cmd.execute_called == 1
        assert self.manager.can_undo() == True
        
        result = self.manager.undo()
        assert cmd.undo_called == 1
        assert result.success == True
        assert self.manager.can_undo() == False
    
    def test_execute_and_redo(self):
        cmd = MockCommand()
        self.manager.execute_command(cmd)
        self.manager.undo()
        
        result = self.manager.redo()
        assert cmd.redo_called == 1
        assert result.success == True
        assert self.manager.can_redo() == False
    
    def test_max_history_limit(self):
        self.manager.state.max_history = 5
        
        for i in range(10):
            cmd = MockCommand()
            self.manager.execute_command(cmd)
        
        # 应该只保留最后 5 个
        assert len(self.manager.state.undo_stack) == 5
    
    def test_clear_history(self):
        for i in range(5):
            cmd = MockCommand()
            self.manager.execute_command(cmd)
        
        self.manager.clear_history()
        assert len(self.manager.state.undo_stack) == 0
        assert len(self.manager.state.redo_stack) == 0
```

## 8. 边界情况处理

| 边界情况 | 处理方案 |
|---------|---------|
| 同时操作多个节点 | 使用 MacroCommand 包装为一个历史条目 |
| 操作失败时 | 不添加到历史记录 |
| 历史记录数量限制 | FIFO 策略，超过时删除最早的记录 |
| 保存/加载历史 | 可选功能，默认不保存（避免文件过大） |
| 项目切换 | 清空当前历史记录 |
| 不可撤销的操作 | 设置 command.can_undo = False |
